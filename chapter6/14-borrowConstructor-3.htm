borrowing a constructor

from book Object Oriented Javascript pdf version, page 210

function Shape(id) {
	this.id = id;
}

function Triangle() {
	Shape.apply(this, arguments);
}
Triangle.prototype = new Shape(101);

var t = new Triangle(202);
console.log(t.id);
console.log(t.__proto__.id);
console.log(delete t.id);
console.log(t.id);
