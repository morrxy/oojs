function's call and apply method

from book Object Oriented Javascript pdf version, page 134

var som_obj = {
	name: 'Ninja',
	say: function(who) {
		return 'Haya ' + who + ', I am a ' + this.name;
	}
};

console.log(som_obj.say('Dude'));

var my_obj = {
	name: 'Scripting guru'
};

console.log(som_obj.say.call(my_obj, 'Dude'));
