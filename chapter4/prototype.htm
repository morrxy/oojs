function's prototype

from book Object Oriented Javascript pdf version, page 133

	var some_obj = {
		name: 'Ninja',
		say: function() {
			return 'I am a ' + this.name;
		}
	}

	function F() {}
	console.log(typeof F.prototype);

	F.prototype = some_obj;

	var obj = new F();
	console.log(obj.name);
	console.log(obj.say());
