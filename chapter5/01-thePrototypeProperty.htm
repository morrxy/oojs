the prototype property

prototype is a property that gets created as soon as you define the function. Its initial value is an empty object.

from book Object Oriented Javascript pdf version, page 167

function foo(a, b) {
	return a * b;
}
console.log(foo.length);
console.log(foo.constructor);
console.log(typeof foo.prototype);
console.log(foo.prototype);
foo.prototype = {};
console.log(foo.prototype);
