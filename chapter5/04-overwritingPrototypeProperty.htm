overwriteing prototype's property with own property

from book Object Oriented Javascript pdf version, page 171

function Gadget(name) {
	this.name = name;
}
Gadget.prototype.name = 'foo';

var toy = new Gadget('camera');
console.log(toy.name);
delete toy.name;
console.log(toy.name);
toy.name = 'camera';
console.log(toy.name);
